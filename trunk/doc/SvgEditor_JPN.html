Classpublic class SvgEditor
InheritanceSvgEditor flash.display.Sprite

使い方：

var editor:SvgEditor = new SvgEditor();
editor.appStage = this;
addChild( editor );

//新規ドキュメント作成
editor.newDocument( 300 , 300 );

又は、

//Svg を読み込んで表示
var svg:XML = <svg>..svg document..</svg>;
editor.parse( svg );
